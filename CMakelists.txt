cmake_minimum_required(VERSION 2.8.18)

project(gomoku)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY $<1:${CMAKE_SOURCE_DIR}/bin>)
# set(CMAKE_BINARY_DIR ${CMAKE_SOURCE_DIR}/bin)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(EXECUTABLE_OUTPUT_PATH ${CMAKE_BINARY_DIR})
set(LIBRARY_OUTPUT_PATH ${CMAKE_BINARY_DIR})


add_definitions("-Ofast") #-Wall -Werror -Wextra


find_package(SDL2 REQUIRED)
include_directories(${PROJECT_SOURCE_DIR}/includes ${SDL2_INCLUDE_DIRS})

file(GLOB_RECURSE gomoku_SOURCES CONFIGURE_DEPENDS "src/*.cpp")

add_executable(gomoku ${gomoku_SOURCES})
target_link_libraries(gomoku ${SDL2_LIBRARIES})
